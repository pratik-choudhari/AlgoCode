# Python program to find least multiple of a digit using 0's and 9's only  
from queue import Queue 
  
# Generate all combinations of 0 and 9  
def generateCombinations(): 
    global inputs 
      
    # Empty queue 
    q = Queue() 
          
    # enque the first digit 9
    q.put("9")  

    for num in range(MAX_COUNT, -1, -1): 
        l1 = q.queue[0]  
        q.get() 
          
        inputs.append(l1)  
          
        l2 = l1  
          
        # Append "0" to l1 and enqueue it 
        l1 += "0"
        q.put(l1)  
          
        # Append "9" to l2 and enqueue it. 
        l2 += "9"
        q.put(l2) 
  
  
def findLeastMultiple(n): 
    global inputs 
    if "9" in inputs:
      inputs.remove("9")
      
    # find the least multiple of n 
    for i in range(len(inputs)): 
        if (int(inputs[i]) % n == 0):  
            return inputs[i]  
  
 
# Maximum digit of numbers
MAX_COUNT = 100000
